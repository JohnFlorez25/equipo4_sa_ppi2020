USE bt69jg4rjgesrg0cgunt;

CREATE TABLE TIPO_USUARIO_P(
CODIGO INT PRIMARY KEY NOT NULL,
TIPO VARCHAR(50) NOT NULL
);


CREATE TABLE USUARIO_P(
CORREO VARCHAR(50) PRIMARY KEY NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
EDAD INT NOT NULL,
CONTRASENA VARCHAR(50) NOT NULL,
RAZA VARCHAR(50) NOT NULL,
TIPO_USUARIO INT NOT NULL,
FOREIGN KEY (TIPO_USUARIO) REFERENCES TIPO_USUARIO_P(CODIGO)
);

CREATE TABLE EVIDENCIA_P(
CODIGO_EV INT PRIMARY KEY NOT NULL,
VIDEO VARCHAR(100) NOT NULL,
CODIGO_USUARIO VARCHAR(50) NOT NULL,
FOREIGN KEY(CODIGO_USUARIO) REFERENCES USUARIO_P(CORREO)
);

CREATE TABLE TIPO_ENTRENAMIENTO_P(
 CODIGO INT PRIMARY KEY NOT NULL,
 TIPO VARCHAR(20) NOT NULL
 );

CREATE TABLE ENTRENAMIENTO_P(
 CODIGO INT PRIMARY KEY NOT NULL,
 CONTENIDO VARCHAR(50) NOT NULL,
 TIPO_ENTRE INT NOT NULL,
 FOREIGN KEY(TIPO_ENTRE) REFERENCES TIPO_ENTRENAMIENTO_P(CODIGO)
);

CREATE TABLE ENTRENAMIENTO_USUARIO_P(
  CODIGO_USUARIO VARCHAR(50) NOT NULL,
  CODIGO_ENTRENAMIENTO INT NOT NULL,
  PRIMARY KEY(CODIGO_USUARIO,CODIGO_ENTRENAMIENTO),
  FOREIGN KEY (CODIGO_USUARIO) REFERENCES USUARIO_P(CORREO),
  FOREIGN KEY(CODIGO_ENTRENAMIENTO) REFERENCES ENTRENAMIENTO_P(CODIGO)
 );
